MessageBox 静态方法示例{alert和confirm方法}

   

MessageBox.alert({ 
	icon:MB_ICON.SUCCESS, 
	caption:'这是个示例的messagebox',
	content:'如果您的浏览器3秒内没有自动跳转，请点击这里' 
});
MessageBox.confirm({ 
	caption:'这是个示例的messagebox', 
	content:'如果您的浏览器3秒内没有自动跳转，请点击这里' 
});

代码示例

new MessageBox({ 
	caption: '这是个示例的messagebox',
	content: '如果您的浏览器3秒内没有自动跳转，请点击这里',
	icon: MB_ICON.SUCCESS,
	buttons: [MB_BUTTON.OK, MB_BUTTON.CANCEL:],
	focusButton: MB_BUTTON.OK
	modal: true,
	lazyShow: false,
	onclose: function(e) {
		alert(e.retVal);
	}
});
